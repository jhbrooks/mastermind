def human_mark(valid_options)
  puts "Please select a mark for slot #{state.current_slot}."
  mark = gets.chomp!.upcase
  if !(valid_options.include?(mark))
    puts "Invalid entry! Please try again."
    human_mark(valid_options)
  else
    return mark
  end
end
